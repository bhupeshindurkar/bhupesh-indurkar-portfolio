How to Use

HTML Classes:

<!-- Typewriter Effect -->
<span class="typewriter-effect" data-speed="100" data-delay="500">Your text</span>

<!-- Trip Rider (Text Rotation) -->
<span class="trip-rider-typing" data-words="Text1,Text2,Text3" data-speed="2000"></span>

<!-- Wave Text -->
<span class="wave-text">WAVE TEXT</span>

<!-- Neon Effect -->
<span class="neon-text">NEON TEXT</span>

<!-- Hologram Effect -->
<span class="hologram-text">HOLOGRAM</span>

<!-- Scramble Effect -->
<span class="scramble-text" data-trip-effect="scramble">SCRAMBLE</span>

<!-- Glitch Effect -->
<span data-trip-effect="glitch">GLITCH</span>

<!-- Matrix Background -->
<div data-trip-effect="matrix">Content with matrix background</div>
